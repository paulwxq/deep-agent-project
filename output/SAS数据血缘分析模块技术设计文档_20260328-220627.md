# SAS 数据血缘分析模块技术设计文档

## 1. 项目概述与目标

### 1.1 项目背景

在企业数据分析环境中，SAS 代码被广泛用于数据处理和转换。随着数据管道复杂度增加，追踪数据从源表到目标表的流转关系变得至关重要。本项目旨在设计一个 Python 应用模块，能够自动分析 SAS 代码和数据集，生成数据血缘对照表。

### 1.2 项目目标

设计并实现一个 Python 模块，用于：
- 分析 SAS 代码中的数据处理逻辑
- 识别源表（mainframe SAS 数据集、CSV/Excel 文件）与目标表（CSV 输出）之间的字段映射关系
- 生成数据血缘对照表，包含：源表名/文件名、源表字段、目标表名/文件名、目标表字段、转换公式
- 支持从输出表作为分析入口，逆向追踪每个结果表字段对应的源表字段

### 1.3 核心需求

1. **输出文件类型**：所有 SAS 代码的输出文件均为 CSV 格式
2. **源数据类型**：
   - 大部分：映射到 mainframe 的 SAS 数据集（通过 libname 映射）
   - 小部分：CSV 或 Excel 文件
3. **中间文件类型**：SAS 临时文件（.sas7bdat 格式）
4. **处理约束**：由于 SAS 代码和文件数量较多，无法一次性全部提交给 LLM，需要支持分批次/增量处理策略

### 1.4 技术栈

- **Python 版本**：3.12
- **LLM Provider**：可配置（支持 Anthropic、OpenAI、Google 等）
- **框架**：langchain-ai
- **Agent SDK**：deep agents（用于复杂任务规划和子代理协作）

```python
from langchain.chat_models import init_chat_model
```

```python
    def add_record(self, record: LineageRecord):
        """添加血缘记录"""
        self.records.append(record)
    
    def add_records(self, records: List[LineageRecord]):

        self.records.extend(records)

        """导出为 CSV 文件"""

            # 写入表头
            writer.writerow([
                'output_table', 'output_column',
                'original_table', 'original_column',
                'formula', 'script_file'

            # 写入数据
            for record in self.records:
                writer.writerow([
                    record.output_table,
                    record.output_column,
                    record.original_table,
                    record.original_column,
                    record.formula,
                    record.script_file
                ])
        return self.output_path
```

### 4.2 SAS 代码解析策略

#### 4.2.1 表引用识别

```python
import re

class SASTableExtractor:
    """SAS 表引用提取器"""
    
    # 匹配 DATA 语句
    DATA_PATTERN = re.compile(r'DATA\s+(\w+)\s*;', re.IGNORECASE)
    
    # 匹配 SET 语句（单表输入，支持宏变量和括号内选项）
    # 支持格式：SET libname.table、SET &macro_var.、SET table(RENAME=(...))
    SET_PATTERN = re.compile(
        r'SET\s+([&\w]+\.[&\w]+|[&\w]+)(?:\s*\([^)]*\))?\s*;',
        re.IGNORECASE
    )
    
    # 匹配 MERGE 语句（多表合并）
    MERGE_PATTERN = re.compile(r'MERGE\s+([\w\s(IN=)]+)\s*;', re.IGNORECASE)
    
    # 匹配 PROC IMPORT
    PROC_IMPORT_PATTERN = re.compile(
        r'PROC\s+IMPORT\s+DATAFILE\s*=\s*"([^"]+)"\s+OUT\s*=\s*(\w+)',
        re.IGNORECASE
    )
    
    # 匹配 PROC EXPORT
    PROC_EXPORT_PATTERN = re.compile(
        r'PROC\s+EXPORT\s+DATA\s*=\s*(\w+)\s+OUTFILE\s*=\s*"([^"]+)"',
        re.IGNORECASE
    )
    
    # 匹配 libname 语句
    LIBNAME_PATTERN = re.compile(
        r'libname\s+(\w+)\s+"([^"]+)"',
        re.IGNORECASE
    )
    
    def extract_tables(self, sas_code: str) -> Dict[str, List[str]]:
        """提取 SAS 代码中的表引用"""
        result = {
            'input_tables': [],
            'output_tables': [],
            'libname_mappings': {}
        }
        
        # 提取 libname 映射
        for match in self.LIBNAME_PATTERN.finditer(sas_code):
            libref, path = match.groups()
            result['libname_mappings'][libref] = path
        
        # 提取输入表
        for match in self.SET_PATTERN.finditer(sas_code):
            table_name = match.group(1)
            result['input_tables'].append(table_name)
        
        # 提取 MERGE 表
        for match in self.MERGE_PATTERN.finditer(sas_code):
            merge_clause = match.group(1)
            # 解析多个表名
            tables = re.findall(r'(\w+)', merge_clause)
            result['input_tables'].extend(tables)
        
        # 提取输出表
        for match in self.DATA_PATTERN.finditer(sas_code):
            table_name = match.group(1)
            result['output_tables'].append(table_name)
        
        # 提取 PROC EXPORT 输出
        for match in self.PROC_EXPORT_PATTERN.finditer(sas_code):
            data_table, output_file = match.groups()
            result['output_tables'].append(data_table)
            result['output_file'] = output_file
        
        return result
```

#### 4.2.2 字段映射识别

```python
class SASFieldExtractor:
    """SAS 字段映射提取器"""
    
    # 匹配 RENAME 语句（支持跨行、宏变量、函数嵌套）
    # 使用 re.DOTALL 标志使 . 匹配换行符
    RENAME_PATTERN = re.compile(
        r'RENAME\s*=\s*\((.+?)\)',
        re.IGNORECASE | re.DOTALL
    )
    
    # 匹配 KEEP 语句（支持多行）
    KEEP_PATTERN = re.compile(r'KEEP\s+([\w\s]+?);', re.IGNORECASE | re.DOTALL)
    
    # 匹配赋值语句（计算字段）
    ASSIGNMENT_PATTERN = re.compile(
        r'(\w+)\s*=\s*([^;]+)\s*;',
        re.IGNORECASE
    )
    
    def extract_rename_mappings(self, sas_code: str) -> List[Dict[str, str]]:
        """提取重命名映射
        
        支持以下复杂场景：
        1. 宏变量引用：RENAME=(old_&var. = new_field)
        2. 函数嵌套：RENAME=(field1 = func(field2))
        3. 跨行格式：RENAME=(field1 = new1
                             field2 = new2)
        
        Args:
            sas_code: SAS 代码字符串
            
        Returns:
            重命名映射列表
        """
        mappings = []
        for match in self.RENAME_PATTERN.finditer(sas_code):
            rename_clause = match.group(1)
            
            # 处理宏变量展开（简化处理：保留宏变量标记）
            # 实际实现中可集成宏处理器进行展开
            rename_clause = self._expand_macro_variables(rename_clause)
            
            # 解析 "old_name = new_name" 格式
            # 支持字段名包含宏变量、函数嵌套
            pairs = re.findall(
                r'([&\w\.]+(?:\s*\([^)]*\))?)\s*=\s*([&\w\.]+(?:\s*\([^)]*\))?)',
                rename_clause
            )
            for source, target in pairs:
                mappings.append({
                    'source_field': source.strip(),
                    'target_field': target.strip(),
                    'mapping_type': 'rename'
                })
        return mappings
    
    def _expand_macro_variables(self, text: str) -> str:
        """展开宏变量引用（简化版本）
        
        实际实现中应集成 SAS 宏处理器，这里仅做标记保留。
        
        Args:
            text: 包含宏变量的文本
            
        Returns:
            处理后的文本
        """
        # 保留宏变量标记，后续处理时再展开
        # 例如：&lib..churn_raw -> {MACRO:lib}.churn_raw
        return re.sub(r'&(\w+)\.', r'{MACRO:\1}.', text)
    
    def extract_keep_fields(self, sas_code: str) -> List[str]:
        """提取 KEEP 语句指定的输出字段列表
        
        KEEP 语句用于指定 DATA 步输出数据集中保留的字段。
        示例：
            KEEP CustomerId Geography Gender Age Balance credit_scr est_sal;
        
        Args:
            sas_code: SAS 代码字符串
            
        Returns:
            字段名列表
        """
        keep_fields = []
        for match in self.KEEP_PATTERN.finditer(sas_code):
            keep_clause = match.group(1)
            # 解析字段名（支持空格、换行分隔）
            fields = re.findall(r'\b([A-Z_]\w*)\b', keep_clause, re.IGNORECASE)
            # 过滤掉 SAS 关键字
            keywords = {'AND', 'OR', 'NOT', 'IN', 'EQ', 'NE', 'GT', 'LT', 'GE', 'LE'}
            fields = [f for f in fields if f.upper() not in keywords]
            keep_fields.extend(fields)
        return keep_fields
    
    def extract_calculated_fields(self, sas_code: str) -> List[Dict]:
        """提取计算字段"""
        calculated = []
        for match in self.ASSIGNMENT_PATTERN.finditer(sas_code):
            target_field, formula = match.groups()
            # 提取公式中引用的字段
            source_fields = re.findall(r'\b([A-Z]\w+)\b', formula, re.IGNORECASE)
            # 过滤掉数字和关键字
            source_fields = [f for f in source_fields if not f.isdigit()]
            calculated.append({
                'target_field': target_field,
                'source_fields': source_fields,
                'formula': formula.strip(),
                'mapping_type': 'calculated'
            })
        return calculated
```

#### 4.2.3 Mainframe 数据集定义文件解析

Mainframe 数据集定义文件（通常命名为 `def_*.sas`）用于定义从 mainframe 导入的原始数据集的字段结构。这类文件使用 `infile` 语句指定输入文件路径和格式，使用 `input` 语句定义字段名、起始位置和格式。

**示例文件结构（def_Churn.sas）：**

```sas
data &lib.. churn_raw /nolist;
    infile "&source_path/churn_raw&file_sfx" recfm=F lrecl=657;
    input
        @1    CustomerId                       PK2.0
        @3    Surname                           PK2.0
        @5    CreditScore                             PK3.0
        @8    Geography                              S370FPD2.0
        @10   Gender                       $EBCDIC3.
        @13   Age                  S370FPD6.0
        @19   Tenure                   S370FPD6.0
        @25   Balance               S370FPD7.0
        @32   NumOfProducts               S370FPD7.0
        @39   HasCrCard               PK1.0
        @40   IsActiveMember               PK1.0
        @41   EstimagteSalary                   S370FPD7.0
        @48   Exited                    S370FPD7.0
    ;
run;
```

**解析策略：**

```python
class MainframeDefParser:
    """Mainframe 数据集定义文件解析器"""
    
    # 匹配 INFILE 语句
    INFILE_PATTERN = re.compile(
        r'INFILE\s+"([^"]+)"\s*(?:RECFM\s*=\s*(\w+))?\s*(?:LRECL\s*=\s*(\d+))?',
        re.IGNORECASE
    )
    
    # 匹配 INPUT 语句块（多行）
    INPUT_PATTERN = re.compile(
        r'INPUT\s+(.*?)\s*;',
        re.IGNORECASE | re.DOTALL
    )
    
    # 匹配单个字段定义（带位置指针）
    FIELD_PATTERN = re.compile(
        r'@(\d+)\s+([a-zA-Z_][\w\s\']*)\s+([A-Z0-9$_]+(?:\.\d+(?:\.\d+)?)?)',
        re.IGNORECASE
    )
    
    def parse_def_file(self, file_path: str) -> Dict:
        """解析 mainframe 定义文件
        
        Args:
            file_path: def_*.sas 文件路径
            
        Returns:
            解析结果，包含：
            - dataset_name: 数据集名
            - source_file: 源文件路径
            - recfm: 记录格式
            - lrecl: 记录长度
            - fields: 字段列表（字段名、起始位置、格式）
        """
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        result = {
            'dataset_name': self._extract_dataset_name(content),
            'source_file': None,
            'recfm': None,
            'lrecl': None,
            'fields': []
        }
        
        # 提取 INFILE 信息
        infile_match = self.INFILE_PATTERN.search(content)
        if infile_match:
            result['source_file'] = infile_match.group(1)
            result['recfm'] = infile_match.group(2)
            result['lrecl'] = int(infile_match.group(3)) if infile_match.group(3) else None
        
        # 提取 INPUT 字段定义
        input_match = self.INPUT_PATTERN.search(content)
        if input_match:
            input_block = input_match.group(1)
            for field_match in self.FIELD_PATTERN.finditer(input_block):
                position = int(field_match.group(1))
                field_name = field_match.group(2).strip().strip("'\"")
                field_format = field_match.group(3)
                
                result['fields'].append({
                    'field_name': field_name,
                    'position': position,
                    'format': field_format,
                    'field_type': 'numeric' if '$' not in field_format else 'character'
                })
        
        return result
    
    def _extract_dataset_name(self, content: str) -> str:
        """提取数据集名称"""
        # 匹配 DATA 语句
        data_pattern = re.compile(
            r'DATA\s+(&lib\.\.)?\s*([a-zA-Z_]\w*)',
            re.IGNORECASE
        )
        match = data_pattern.search(content)
        if match:
            return match.group(2)
        return None
    
    def build_field_metadata(self, def_file_path: str) -> Dict[str, List[str]]:
        """构建字段元数据缓存
        
        解析所有 def_*.sas 文件，建立数据集名到字段列表的映射。
        
        Args:
            def_file_path: def_*.sas 文件路径或目录
            
        Returns:
            字典：{数据集名：[字段列表]}
        """
        field_metadata = {}
        
        # 如果是目录，查找所有 def_*.sas 文件
        if Path(def_file_path).is_dir():
            def_files = list(Path(def_file_path).glob('def_*.sas'))
        else:
            def_files = [Path(def_file_path)]
        
        for def_file in def_files:
            try:
                result = self.parse_def_file(str(def_file))
                if result['dataset_name']:
                    field_metadata[result['dataset_name']] = [
                        f['field_name'] for f in result['fields']
                    ]
            except Exception as e:
                print(f"解析定义文件失败 {def_file}: {e}")
        
        return field_metadata
```

**使用场景：**

1. **源头字段验证**：当血缘追踪到达 mainframe 数据集时，使用定义文件验证字段是否存在
2. **字段类型推断**：根据定义文件中的格式（如 `$EBCDIC3.` 表示字符型，`S370FPD7.0` 表示数值型）推断字段类型
3. **字段位置映射**：记录字段在原始文件中的字节位置，支持物理层面的血缘追踪

```python
class LineageTracker:
```

### 4.3 血缘追踪算法

```python
class LineageTracker:
    """血缘追踪器"""
    
    def __init__(self):
        # 表到字段的映射缓存
        self.table_field_cache: Dict[str, List[str]] = {}
        # 脚本到表的映射
        self.script_table_map: Dict[str, Dict] = {}
        # 最终血缘记录
        self.lineage_records: List[LineageRecord] = []
    
    def trace_field_lineage(
        self,
        output_table: str,
        output_field: str,
        script_file: str
    ) -> LineageRecord:
        """追踪单个输出字段的血缘
        
        支持 MERGE 多源场景处理：
        1. 读取 SAS 代码中的 BY 语句确定合并键
        2. 对于非合并键字段，根据 KEEP 语句中字段的来源确定原始表
        3. 如果无法确定来源，标记为 "ambiguous"
        """
        
        # Step 1: 解析当前脚本
        if script_file not in self.script_table_map:
            self.script_table_map[script_file] = self._parse_script(script_file)
        
        script_info = self.script_table_map[script_file]
        
        # Step 1.5: 检查是否为 MERGE 场景，提取 BY 语句和源表归属信息
        merge_info = self._extract_merge_info(script_file)
        
        # Step 2: 确定字段映射类型
        mapping_info = self._find_field_mapping(
            output_field,
            script_info['field_mappings']
        )
        
        # Step 3: 递归追踪源表字段
        source_tables = []
        source_fields = []
        
        if mapping_info['mapping_type'] == 'passthrough':
            # 直通字段，直接从输入表继承
            # MERGE 场景：根据字段来源确定原始表
            if merge_info and output_field in merge_info['field_source_map']:
                # 有明确的源表归属
                source_table = merge_info['field_source_map'][output_field]
                if source_table == 'ambiguous':
                    # 无法确定来源，使用所有输入表
                    for input_table in script_info['input_tables']:
                        source_tables.append(input_table)
                        source_fields.append(output_field)
                else:
                    # 有明确的源表
                    source_tables.append(source_table)
                    source_fields.append(output_field)
            else:
                # 非 MERGE 场景或无归属信息
                for input_table in script_info['input_tables']:
                    source_tables.append(input_table)
                    source_fields.append(output_field)
        
        elif mapping_info['mapping_type'] == 'rename':
            # 重命名字段，追踪原始字段名
            source_field = mapping_info['source_field']
            # MERGE 场景：检查源字段的表归属
            if merge_info and source_field in merge_info['field_source_map']:
                source_table = merge_info['field_source_map'][source_field]
                if source_table == 'ambiguous':
                    for input_table in script_info['input_tables']:
                        source_tables.append(input_table)
                        source_fields.append(source_field)
                else:
                    source_tables.append(source_table)
                    source_fields.append(source_field)
            else:
                for input_table in script_info['input_tables']:
                    source_tables.append(input_table)
                    source_fields.append(source_field)
        
        elif mapping_info['mapping_type'] == 'calculated':
            # 计算字段，追踪所有依赖字段
            source_fields = mapping_info['source_fields']
            # MERGE 场景：为每个依赖字段确定源表
            for src_field in source_fields:
                if merge_info and src_field in merge_info['field_source_map']:
                    source_table = merge_info['field_source_map'][src_field]
                    if source_table == 'ambiguous':
                        # 无法确定来源，记录所有输入表
                        for input_table in script_info['input_tables']:
                            source_tables.append(input_table)
                    else:
                        source_tables.append(source_table)
                else:
                    # 非 MERGE 场景或无归属信息
                    for input_table in script_info['input_tables']:
                        source_tables.append(input_table)
        
        # Step 4: 递归处理中间表
        final_source_tables = []
        final_source_fields = []
        
        for src_table, src_field in zip(source_tables, source_fields):
            if self._is_intermediate_table(src_table):
                # 递归追踪中间表
                parent_script = self._find_script_producing_table(src_table)
                if parent_script:
                    recursive_record = self.trace_field_lineage(
                        src_table, src_field, parent_script
                    )
                    final_source_tables.append(recursive_record.original_table)
                    final_source_fields.append(recursive_record.original_column)
                else:
                    final_source_tables.append(src_table)
                    final_source_fields.append(src_field)
            else:
                final_source_tables.append(src_table)
                final_source_fields.append(src_field)
        
        # Step 5: 构建血缘记录
        return LineageRecord(
            output_table=output_table,
            output_column=output_field,
            original_table='; '.join(set(final_source_tables)),
            original_column='; '.join(final_source_fields),
            formula=mapping_info.get('formula', 'passthrough'),
            script_file=script_file
        )
    
    def _is_intermediate_table(self, table_name: str) -> bool:
        """判断是否为中间表（.sas7bdat 文件）
        
        中间表特征：
        1. 不是 mainframe 数据集（没有 libname 映射）
        2. 不是原始 CSV/Excel 文件
        3. 通常是临时生成的 SAS 数据集
        
        Args:
            table_name: 表名
            
        Returns:
            True 如果是中间表
        """
        # 检查是否是已知的源文件类型
        if table_name.endswith('_raw') or table_name.endswith('_input'):
            return False
        
        # 检查是否在 mainframe 定义文件中有定义
        if table_name in self.field_metadata:
            return False
        
        # 检查是否存在对应的 CSV/Excel 文件
        for ext in ['.csv', '.xlsx', '.xls']:
            if Path(f"{self.data_dir}/{table_name}{ext}").exists():
                return False
        
        # 默认认为是中间表（需要进一步追踪）
        return True
    
    def _find_script_producing_table(self, table_name: str) -> Optional[str]:
        """查找生成指定表的 SAS 脚本文件
        
        定位策略：
        1. 从输出 CSV 文件名反推：如果 table_name 是 churn_out，查找输出 churn_out.csv 的脚本
        2. 扫描所有 SAS 脚本的 DATA 语句，找到生成该表的脚本
        3. 使用缓存避免重复扫描
        
        Args:
            table_name: 表名
            
        Returns:
            SAS 脚本文件路径，如果找不到则返回 None
        """
        # 检查缓存
        if hasattr(self, '_table_script_cache'):
            if table_name in self._table_script_cache:
                return self._table_script_cache[table_name]
        else:
            self._table_script_cache = {}
        
        # 策略 1: 从输出 CSV 文件名反推
        # 假设表名 churn_out 对应脚本 01_churn_transform.sas
        # 通过扫描所有脚本的 PROC EXPORT 语句来建立映射
        sas_files = list(Path(self.sas_code_dir).glob('*.sas'))
        
        for sas_file in sas_files:
            with open(sas_file, 'r', encoding='utf-8') as f:
                content = f.read()
            
            # 检查 DATA 语句是否生成该表
            data_pattern = re.compile(
                r'DATA\s+([a-zA-Z_]\w*(?:\.[a-zA-Z_]\w*)?)\s*;',
                re.IGNORECASE
            )
            for match in data_pattern.finditer(content):
                generated_table = match.group(1)
                # 处理 libref.table 格式
                if '.' in generated_table:
                    generated_table = generated_table.split('.', 1)[1]
                
                if generated_table == table_name:
                    # 确认该脚本确实输出这个表（检查 PROC EXPORT）
                    export_pattern = re.compile(
                        r'PROC\s+EXPORT\s+DATA\s*=\s*([a-zA-Z_]\w*)',
                        re.IGNORECASE
                    )
                    for export_match in export_pattern.finditer(content):
                        export_table = export_match.group(1)
                        if export_table == table_name:
                            self._table_script_cache[table_name] = str(sas_file)
                            return str(sas_file)
                    
                    # 即使没有 PROC EXPORT，也认为该脚本生成此表
                    self._table_script_cache[table_name] = str(sas_file)
                    return str(sas_file)
        
        # 策略 2: 如果是中间表（.sas7bdat），查找生成它的脚本
        # 中间表通常是上一个脚本的 DATA 步输出
        # 通过扫描所有脚本的 DATA 语句来定位
        for sas_file in sas_files:
            with open(sas_file, 'r', encoding='utf-8') as f:
                content = f.read()
            
            # 提取所有 DATA 语句生成的表
            data_pattern = re.compile(
                r'DATA\s+([a-zA-Z_]\w*)\s*;',
                re.IGNORECASE
            )
            for match in data_pattern.finditer(content):
                generated_table = match.group(1)
                if generated_table == table_name:
                    self._table_script_cache[table_name] = str(sas_file)
                    return str(sas_file)
        
        # 找不到生成该表的脚本
        return None
    
    def _parse_script(self, script_file: str) -> Dict:
        """解析 SAS 脚本文件
        
        Args:
            script_file: SAS 脚本文件路径
            
        Returns:
            解析结果字典，包含：
            - input_tables: 输入表列表
            - output_tables: 输出表列表
            - field_mappings: 字段映射列表
            - output_fields: KEEP 语句指定的输出字段
        """
        analyzer = SASAnalyzerTool()
        result = analyzer.analyze_file(script_file)
        
        return {
            'input_tables': [t.table_name for t in result.input_tables],
            'output_tables': [t.table_name for t in result.output_tables],
            'field_mappings': result.field_mappings,
            'output_fields': result.output_fields
        }
    
    def _extract_merge_info(self, script_file: str) -> Optional[Dict]:
        """提取 MERGE 语句的合并信息
        
        解析 MERGE 场景下的：
        1. BY 语句中的合并键
        2. 各源表的 IN= 标记
        3. KEEP 语句中字段的来源表归属
        
        Args:
            script_file: SAS 脚本文件路径
            
        Returns:
            MERGE 信息字典，包含：
            - merge_keys: 合并键列表（BY 语句字段）
            - source_tables: 参与合并的源表列表
            - field_source_map: 字段到源表的映射（无法确定时为 "ambiguous"）
            如果不是 MERGE 场景，返回 None
        """
        with open(script_file, 'r', encoding='utf-8') as f:
            sas_code = f.read()
        
        # 检查是否存在 MERGE 语句
        merge_pattern = re.compile(
            r'MERGE\s+([\w\s(IN=)]+?)\s*;',
            re.IGNORECASE
        )
        merge_match = merge_pattern.search(sas_code)
        
        if not merge_match:
            return None
        
        # 提取 MERGE 表列表
        merge_clause = merge_match.group(1)
        # 解析表名和 IN= 标记
        table_pattern = re.compile(r'(\w+)(?:\s*\(IN\s*=\s*(\w+)\))?')
        tables = table_pattern.findall(merge_clause)
        source_tables = [t[0] for t in tables]
        in_markers = {t[0]: t[1] for t in tables if t[1]}
        
        # 提取 BY 语句
        by_pattern = re.compile(r'BY\s+([\w\s]+?)\s*;', re.IGNORECASE)
        by_match = by_pattern.search(sas_code)
        merge_keys = []
        if by_match:
            merge_keys = re.findall(r'\b(\w+)\b', by_match.group(1))
        
        # 提取 KEEP 语句
        keep_pattern = re.compile(r'KEEP\s+([\w\s]+?)\s*;', re.IGNORECASE)
        keep_match = keep_pattern.search(sas_code)
        keep_fields = []
        if keep_match:
            keep_fields = re.findall(r'\b(\w+)\b', keep_match.group(1))
        
        # 构建字段到源表的映射
        field_source_map = {}
        
        # 合并键来自所有源表
        for key in merge_keys:
            field_source_map[key] = '; '.join(source_tables)
        
        # 非合并键字段：尝试确定来源
        # 简化策略：如果字段只在一个源表中存在，则归属该表
        # 实际实现中需要查询各源表的字段元数据
        for field in keep_fields:
            if field not in field_source_map:
                # 暂时标记为 ambiguous，等待元数据查询后确定
                field_source_map[field] = 'ambiguous'
        
        return {
            'merge_keys': merge_keys,
            'source_tables': source_tables,
            'in_markers': in_markers,
            'field_source_map': field_source_map
        }
    
    def _find_field_mapping(
        self,
        output_field: str,
        field_mappings: List
    ) -> Dict:
        """查找输出字段的映射信息
        
        Args:
            output_field: 输出字段名
            field_mappings: 字段映射列表
            
        Returns:
            映射信息字典，包含：
            - mapping_type: passthrough/rename/calculated
            - source_field: 源字段名（如果是 rename）
            - source_fields: 源字段列表（如果是 calculated）
            - formula: 转换公式
        """
        # 首先检查是否是计算字段
        for mapping in field_mappings:
            if mapping.target_field == output_field:
                if mapping.mapping_type == 'calculated':
                    return {
                        'mapping_type': 'calculated',
                        'source_fields': mapping.source_field.split('; '),
                        'formula': mapping.formula
                    }
                elif mapping.mapping_type == 'rename':
                    return {
                        'mapping_type': 'rename',
                        'source_field': mapping.source_field,
                        'formula': mapping.formula
                    }
        
        # 如果没有找到明确的映射，假设是 passthrough（直通字段）
        return {
            'mapping_type': 'passthrough',
            'source_field': output_field,
            'formula': 'passthrough'
        }
```

## 5. 上下文处理策略

针对 SAS 代码和文件数量较多的场景，本设计采用多层次的上下文处理策略，确保系统能够稳定处理各种规模的数据。

### 5.1 单文件过大处理策略

当单个 SAS 代码文件超过 LLM 上下文窗口限制时，采用分块读取和流式处理策略：

```python
from typing import Generator, List, Tuple
from dataclasses import dataclass

@dataclass
class CodeChunk:
    """代码块"""
    chunk_id: int
    content: str
    start_line: int
    end_line: int
    is_continuation: bool  # 是否是一个逻辑块的延续

class SASCodeChunker:
    """SAS 代码分块器"""
    
    # SAS 逻辑块起始关键字
    BLOCK_START_KEYWORDS = [
        'DATA ', 'PROC ', '%MACRO ', '%DO ', 'IF ', 'DO ',
        'SELECT ', 'WHEN ', 'OTHERWISE ', 'END;'
    ]
    
    # 最大块大小（字符数）
    MAX_CHUNK_SIZE = 4000  # 预留空间给 LLM prompt
    
    def __init__(self, max_chunk_size: int = MAX_CHUNK_SIZE):
        self.max_chunk_size = max_chunk_size
    
    def chunk_file(self, file_path: str) -> Generator[CodeChunk, None, None]:
        """将 SAS 文件分割为逻辑块
        
        Args:
            file_path: SAS 文件路径
            
        Yields:
            CodeChunk: 代码块
        """
        with open(file_path, 'r', encoding='utf-8') as f:
            lines = f.readlines()
        
        chunk_id = 0
        current_chunk_lines = []
        current_size = 0
        start_line = 1
        
        for i, line in enumerate(lines, 1):
            line_size = len(line)
            
            # 检查是否是新逻辑块的开始
            is_block_start = any(
                line.strip().upper().startswith(kw) 
                for kw in self.BLOCK_START_KEYWORDS
            )
            
            # 如果当前块已满且遇到新块，yield 当前块
            if current_size + line_size > self.max_chunk_size and current_chunk_lines:
                yield CodeChunk(
                    chunk_id=chunk_id,
                    content=''.join(current_chunk_lines),
                    start_line=start_line,
                    end_line=i - 1,
                    is_continuation=False
                )
                chunk_id += 1
                current_chunk_lines = []
                current_size = 0
                start_line = i
            
            # 如果是块开始且当前有内容，先 yield
            if is_block_start and current_chunk_lines and current_size > 0:
                yield CodeChunk(
                    chunk_id=chunk_id,
                    content=''.join(current_chunk_lines),
                    start_line=start_line,
                    end_line=i - 1,
                    is_continuation=False
                )
                chunk_id += 1
                current_chunk_lines = []
                current_size = 0
                start_line = i
            
            current_chunk_lines.append(line)
            current_size += line_size
        
        # yield 最后一个块
        if current_chunk_lines:
            yield CodeChunk(
                chunk_id=chunk_id,
                content=''.join(current_chunk_lines),
                start_line=start_line,
                end_line=len(lines),
                is_continuation=False
            )
    
    def chunk_with_context(
        self, 
        file_path: str, 
        overlap_lines: int = 5
    ) -> Generator[CodeChunk, None, None]:
        """带上下文重叠的分块
        
        Args:
            file_path: SAS 文件路径
            overlap_lines: 重叠行数，用于保持上下文连续性
            
        Yields:
            CodeChunk: 带上下文的代码块
        """
        chunks = list(self.chunk_file(file_path))
        
        for i, chunk in enumerate(chunks):
            # 添加前一个块的尾部作为上下文
            if i > 0:
                prev_chunk = chunks[i - 1]
                context_lines = prev_chunk.content.split('\n')[-overlap_lines:]
                chunk.content = '\n'.join(context_lines) + '\n' + chunk.content
                chunk.start_line = prev_chunk.end_line - overlap_lines + 1
                chunk.is_continuation = True
            
            yield chunk
```

**使用示例：**

```python

    # 对每个块进行分析
```

### 5.2 解析结果过大处理策略

当解析结果过大时，采用分批处理和结果压缩策略：

```python
from typing import Dict, Any, List
import json
from pathlib import Path

class ResultBatchProcessor:
    """结果分批处理器"""
    
    def __init__(self, batch_size: int = 100, temp_dir: str = "/tmp/lineage_batches"):
        self.batch_size = batch_size
        self.temp_dir = Path(temp_dir)
        self.temp_dir.mkdir(parents=True, exist_ok=True)
    
    def process_in_batches(
        self,
        items: List[Any],
        process_fn: callable,
        merge_fn: callable
    ) -> Any:
        """分批处理大量数据
        
        Args:
            items: 待处理的数据项列表
            process_fn: 处理单个批次的函数
            merge_fn: 合并批次结果的函数
            
        Returns:
            合并后的结果
        """
        results = []
        
        for i in range(0, len(items), self.batch_size):
            batch = items[i:i + self.batch_size]
            batch_result = process_fn(batch)
            results.append(batch_result)
            
            # 将中间结果写入临时文件，防止内存溢出
            temp_file = self.temp_dir / f"batch_{i // self.batch_size}.json"
            with open(temp_file, 'w') as f:
                json.dump(batch_result, f)
        
        # 合并所有批次结果
        final_result = merge_fn(results)
        
        # 清理临时文件
        for temp_file in self.temp_dir.glob("batch_*.json"):
            temp_file.unlink()
        
        return final_result


class ResultCompressor:
    """结果压缩器"""
    
    @staticmethod
    def compress_lineage_records(records: List[Dict]) -> Dict:
        """压缩血缘记录，去除冗余信息
        
        Args:
            records: 血缘记录列表
            
        Returns:
            压缩后的结果
        """
        # 按输出表分组
        grouped = {}
        for record in records:
            key = f"{record['output_table']}:{record['script_file']}"
            if key not in grouped:
                grouped[key] = {
                    'output_table': record['output_table'],
                    'script_file': record['script_file'],
                    'output_columns': [],
                    'source_tables': set()
                }
            
            grouped[key]['output_columns'].append({
                'output_column': record['output_column'],
                'original_table': record['original_table'],
                'original_column': record['original_column'],
                'formula': record['formula']
            })
            
            # 收集源表
            for table in record['original_table'].split('; '):
                grouped[key]['source_tables'].add(table)
        
        # 转换为列表并序列化集合
        compressed = []
        for item in grouped.values():
            item['source_tables'] = list(item['source_tables'])
            compressed.append(item)
        
        return {
            'total_records': len(records),
            'unique_output_tables': len(grouped),
            'compressed_data': compressed
        }
    
    @staticmethod
    def extract_summary(result: Dict) -> str:
        """提取结果摘要，用于 LLM 上下文
        
        Args:
            result: 完整结果
            
        Returns:
            摘要字符串
        """
        summary_lines = []
        summary_lines.append(f"总记录数：{result['total_records']}")
        summary_lines.append(f"唯一输出表数：{result['unique_output_tables']}")
        
        for item in result['compressed_data'][:10]:  # 只显示前 10 个
            summary_lines.append(
                f"- {item['output_table']}: "
                f"{len(item['output_columns'])} 字段，"
                f"源表：{', '.join(item['source_tables'][:3])}"
            )
        
        return '\n'.join(summary_lines)
```

### 5.3 递归深度过深处理策略

针对中间表递归追踪可能导致的深度过深问题，采用深度限制和循环检测机制：

```python
from typing import Set, Optional
from dataclasses import dataclass, field
import functools

@dataclass
class RecursionConfig:
    """递归配置"""
    max_depth: int = 10  # 最大递归深度
    timeout_seconds: int = 300  # 超时时间（秒）
    enable_cycle_detection: bool = True

@dataclass
class RecursionState:
    """递归状态"""
    current_depth: int = 0
    visited_tables: Set[str] = field(default_factory=set)
    call_stack: List[str] = field(default_factory=list)

class CycleDetectedError(Exception):
    """循环依赖检测异常"""
    pass

class MaxDepthExceededError(Exception):
    """最大深度超出异常"""
    pass

class SafeRecursiveTracker:
    """安全递归追踪器"""
    
    def __init__(self, config: RecursionConfig = None):
        self.config = config or RecursionConfig()
        self.cache = {}  # 结果缓存
    
    def trace_with_safety(
        self,
        table_name: str,
        script_file: str,
        state: Optional[RecursionState] = None
    ) -> List[LineageRecord]:
        """带安全保护的递归追踪
        
        Args:
            table_name: 表名
            script_file: 脚本文件路径
            state: 递归状态
            
        Returns:
            血缘记录列表
            
        Raises:
            CycleDetectedError: 检测到循环依赖
            MaxDepthExceededError: 超出最大深度
        """
        # 初始化状态
        if state is None:
            state = RecursionState()
        
        # 检查深度限制
        if state.current_depth > self.config.max_depth:
            raise MaxDepthExceededError(
                f"递归深度超出限制 ({self.config.max_depth}): "
                f"{' -> '.join(state.call_stack)}"
            )
        
        # 检测循环依赖
        if self.config.enable_cycle_detection:
            if table_name in state.visited_tables:
                cycle_path = ' -> '.join(
                    state.call_stack + [table_name]
                )
                raise CycleDetectedError(f"检测到循环依赖：{cycle_path}")
        
        # 更新状态
        state.current_depth += 1
        state.visited_tables.add(table_name)
        state.call_stack.append(table_name)
        
        try:
            # 检查缓存
            cache_key = f"{table_name}:{script_file}"
            if cache_key in self.cache:
                return self.cache[cache_key]
            
            # 执行实际追踪逻辑
            records = self._do_trace(table_name, script_file, state)
            
            # 缓存结果
            self.cache[cache_key] = records
            
            return records
            
        finally:
            # 清理状态（回溯）
            state.current_depth -= 1
            state.call_stack.pop()
            # 注意：visited_tables 不清理，用于跨分支循环检测
    
    def _do_trace(
        self,
        table_name: str,
        script_file: str,
        state: RecursionState
    ) -> List[LineageRecord]:
        """实际字段级血缘追踪逻辑
        
        执行步骤：
        1. 读取并解析 SAS 脚本
        2. 调用 SASAnalyzerTool 提取输入表、输出字段列表、字段映射
        3. 对每个输出字段，确定映射类型（passthrough/rename/calculated）
        4. 递归追踪输入表的血缘（如果是中间表）
        5. 构建并返回血缘记录列表
        
        Args:
            table_name: 当前要追踪的表名
            script_file: 生成该表的 SAS 脚本路径
            state: 递归状态
            
        Returns:
            血缘记录列表
        """
        from sas_lineage.tools.sas_analyzer import SASAnalyzerTool
        
        # Step 1: 读取 SAS 脚本
        with open(script_file, 'r', encoding='utf-8') as f:
            sas_code = f.read()
        
        # Step 2: 调用分析工具提取信息
        analyzer = SASAnalyzerTool()
        analysis_result = analyzer.analyze_script(sas_code, script_file)
        
        # analysis_result 包含：
        # - input_tables: List[SASTableRef] 输入表列表
        # - output_tables: List[SASTableRef] 输出表列表
        # - output_fields: List[str] KEEP 语句指定的输出字段列表
        # - field_mappings: List[SASFieldMapping] 字段映射关系
        
        records = []
        
        # Step 3: 对每个输出字段进行血缘追踪
        for output_field in analysis_result.output_fields:
            # 查找该字段的映射信息
            mapping = self._find_field_mapping(output_field, analysis_result.field_mappings)
            
            # Step 4: 根据映射类型确定源字段
            source_tables = []
            source_fields = []
            
            if mapping.mapping_type == 'passthrough':
                # 直通字段：直接从所有输入表继承
                for input_table in analysis_result.input_tables:
                    source_tables.append(input_table.table_name)
                    source_fields.append(output_field)
            
            elif mapping.mapping_type == 'rename':
                # 重命名字段：使用原始字段名
                for input_table in analysis_result.input_tables:
                    source_tables.append(input_table.table_name)
                    source_fields.append(mapping.source_field)
            
            elif mapping.mapping_type == 'calculated':
                # 计算字段：使用公式中引用的所有源字段
                source_fields = mapping.source_fields
                for input_table in analysis_result.input_tables:
                    source_tables.append(input_table.table_name)
            
            # Step 5: 递归处理中间表（.sas7bdat 文件）
            final_source_tables = []
            final_source_fields = []
            
            for src_table, src_field in zip(source_tables, source_fields):
                if self._is_intermediate_table(src_table):
                    # 递归追踪中间表
                    parent_script = self._find_script_producing_table(src_table)
                    if parent_script:
                        recursive_records = self.trace_with_safety(
                            src_table, parent_script, state
                        )
                        # 从递归结果中提取源表信息
                        for rec in recursive_records:
                            final_source_tables.append(rec.original_table)
                            final_source_fields.append(rec.original_column)
                    else:
                        # 找不到生成脚本，直接使用当前表
                        final_source_tables.append(src_table)
                        final_source_fields.append(src_field)
                else:
                    # 源表（mainframe 数据集或外部文件）
                    final_source_tables.append(src_table)
                    final_source_fields.append(src_field)
            
            # Step 6: 构建血缘记录
            record = LineageRecord(
                output_table=table_name,
                output_column=output_field,
                original_table='; '.join(set(final_source_tables)),
                original_column='; '.join(final_source_fields),
                formula=mapping.formula if mapping.mapping_type == 'calculated' else mapping.mapping_type,
                script_file=script_file
            )
            records.append(record)
        
        return records
    
    def trace_with_timeout(
        self,
        table_name: str,
        script_file: str
    ) -> List[LineageRecord]:
        """带超时保护的追踪
        
        Args:
            table_name: 表名
            script_file: 脚本文件路径
            
        Returns:
            血缘记录列表
        """
        import signal
        
        def timeout_handler(signum, frame):
            raise TimeoutError(
                f"递归追踪超时 ({self.config.timeout_seconds}秒)"
            )
        
        # 设置超时
        signal.signal(signal.SIGALRM, timeout_handler)
        signal.alarm(self.config.timeout_seconds)
        
        try:
            return self.trace_with_safety(table_name, script_file)
        finally:
            # 取消超时
            signal.alarm(0)


# 使用示例
def build_lineage_graph(output_table: str, output_script: str):
    """构建血缘图"""
    tracker = SafeRecursiveTracker(
        config=RecursionConfig(
            max_depth=10,
            timeout_seconds=300,
            enable_cycle_detection=True
        )
    )
    
    try:
        records = tracker.trace_with_safety(output_table, output_script)
        return records
    except CycleDetectedError as e:
        print(f"警告：{e}")
        # 返回部分结果或采取其他措施
        return []
    except MaxDepthExceededError as e:
        print(f"警告：{e}")
        # 返回已追踪到的结果
        return []
    except TimeoutError as e:
        print(f"错误：{e}")
        return []
```

### 5.4 综合处理流程

```python
class RobustLineageAnalyzer:
    """健壮的 lineage 分析器（综合所有策略）"""
    
    def __init__(self, config: SASLineageConfig):
        self.config = config
        self.chunker = SASCodeChunker()
        self.batch_processor = ResultBatchProcessor(batch_size=config.batch_size)
        self.recursive_tracker = SafeRecursiveTracker()
    
    def analyze_robust(self, target_output: str) -> List[LineageRecord]:
        """健壮的 lineage 分析
        
        综合应用：
        1. 文件分块（处理大文件）
        2. 分批处理（处理大量结果）
        3. 安全递归（处理深度递归）
        """
        # Step 1: 找到生成目标输出的 SAS 脚本
        script_file = self._find_script_for_output(target_output)
        
        # Step 2: 检查文件大小，决定是否分块
        file_size = Path(script_file).stat().st_size
        if file_size > 50 * 1024:  # 大于 50KB
            # 使用分块策略
            return self._analyze_chunked(target_output, script_file)
        else:
            # 直接分析
            return self._analyze_direct(target_output, script_file)
    
    def _analyze_chunked(self, target_output: str, script_file: str):
        """分块分析大文件"""
        all_records = []
        
        for chunk in self.chunker.chunk_with_context(script_file):
            # 分析每个块
            chunk_records = self._analyze_chunk(chunk, target_output)
            all_records.extend(chunk_records)
        
        # 合并结果
        return self._merge_chunk_results(all_records)
    
    def _analyze_direct(self, target_output: str, script_file: str):
        """直接分析（小文件）"""
        try:
            return self.recursive_tracker.trace_with_safety(
                target_output, script_file
            )
        except (CycleDetectedError, MaxDepthExceededError) as e:
            print(f"警告：追踪异常 - {e}")
            return []
```

### 5.3 增量处理策略

针对大量 SAS 文件的处理，采用分批次策略：

```python
class BatchLineageAnalyzer:
    """批量血缘分析器"""
    
    def __init__(self, config: SASLineageConfig, batch_size: int = 10):
        self.config = config
        self.batch_size = batch_size
        self.agent = create_lineage_agent(config)
    
    def analyze_all(self) -> Path:
        """分析所有 SAS 文件"""
        # Step 1: 发现所有输出 CSV
        output_csvs = self._discover_output_csvs()
        
        # Step 2: 分批次处理
        store = LineageStore(self.config.output_dir / self.config.output_filename)
        
        for i in range(0, len(output_csvs), self.batch_size):
            batch = output_csvs[i:i + self.batch_size]
            self._process_batch(batch, store)
        
        # Step 3: 导出结果
        return store.export_csv()
    
    def _process_batch(self, output_csvs: List[str], store: LineageStore):
        """处理一批输出 CSV"""
        for csv_file in output_csvs:
            # 为每个输出表创建分析任务
            result = self.agent.invoke({
                "messages": [{
                    "role": "user",
                    "content": f"分析输出表 {csv_file} 的血缘关系"
                }]
            })
            # 将结果添加到存储
            store.add_records(result['lineage_records'])
```

### 5.4 错误处理与验证

```python
class LineageValidation:
    """血缘分析验证器"""
    
    @staticmethod
    def validate_record(record: LineageRecord) -> List[str]:
        """验证单条血缘记录"""
        errors = []
        
        # 检查必填字段
        if not record.output_table:
            errors.append("output_table 不能为空")
        if not record.output_column:
            errors.append("output_column 不能为空")
        if not record.original_table:
            errors.append("original_table 不能为空")
        if not record.original_column:
            errors.append("original_column 不能为空")
        if not record.script_file:
            errors.append("script_file 不能为空")
        
        # 检查多源一致性
        original_tables = record.original_table.split('; ')
        original_columns = record.original_column.split('; ')
        if len(original_tables) != len(original_columns):
            errors.append("多源表与字段数量不匹配")
        
        return errors
    
    @staticmethod
    def generate_validation_report(records: List[LineageRecord]) -> str:
        """生成验证报告"""
        valid_count = 0
        invalid_records = []
        
        for record in records:
            errors = LineageValidation.validate_record(record)
            if errors:
                invalid_records.append((record, errors))
            else:
                valid_count += 1
        
        report = f"""血缘分析验证报告
====================
总记录数：{len(records)}
有效记录：{valid_count}
无效记录：{len(invalid_records)}

无效记录详情：
"""
        for record, errors in invalid_records:
            report += f"\n- {record.output_table}.{record.output_column}:\n"
            for error in errors:
                report += f"  * {error}\n"
        
        return report
```

### 8.1 文件发现工具 (file_discovery.py)

```python
from pathlib import Path
from typing import List, Dict, Optional
from dataclasses import dataclass
import fnmatch

@dataclass
class DiscoveredFile:
    """发现的文件"""
    path: Path
    relative_path: str
    file_type: str  # 'sas_script', 'output_csv', 'source_csv', 'sas7bdat'
    size_bytes: int

class FileDiscoveryTool:
    """文件发现工具"""
    
    def __init__(self, root_dir: str):
        self.root_dir = Path(root_dir)
    
    def find_sas_scripts(
        self, 
        subdir: Optional[str] = None,
        pattern: str = "*.sas"
    ) -> List[DiscoveredFile]:
        """查找所有 SAS 脚本文件
        
        Args:
            subdir: 可选的子目录
            pattern: 文件匹配模式
            
        Returns:
            发现的文件列表
        """
        search_dir = self.root_dir / subdir if subdir else self.root_dir
        files = []
        
        for path in search_dir.rglob(pattern):
            if path.is_file():
                rel_path = path.relative_to(self.root_dir)
                files.append(DiscoveredFile(
                    path=path,
                    relative_path=str(rel_path),
                    file_type='sas_script',
                    size_bytes=path.stat().st_size
                ))
        
        # 按文件大小排序，小文件优先
        files.sort(key=lambda f: f.size_bytes)
        return files
    
    def find_output_csv(
        self,
        pattern: str = "*_out.csv"
    ) -> List[DiscoveredFile]:
        """查找所有输出 CSV 文件"""
        files = []
        
        for path in self.root_dir.rglob(pattern):
            if path.is_file() and path.suffix == '.csv':
                # 确认是输出文件（不是源文件）
                if '_out' in path.stem or 'output' in path.stem.lower():
                    rel_path = path.relative_to(self.root_dir)
                    files.append(DiscoveredFile(
                        path=path,
                        relative_path=str(rel_path),
                        file_type='output_csv',
                        size_bytes=path.stat().st_size
                    ))
        
        return files
    
    def find_source_files(
        self,
        extensions: List[str] = ['.csv', '.xlsx', '.xls']
    ) -> List[DiscoveredFile]:
        """查找源数据文件"""
        files = []
        
        for ext in extensions:
            for path in self.root_dir.rglob(f"*{ext}"):
                if path.is_file():
                    # 排除输出文件
                    if '_out' not in path.stem and 'output' not in path.stem.lower():
                        rel_path = path.relative_to(self.root_dir)
                        file_type = 'source_csv' if ext == '.csv' else 'source_excel'
                        files.append(DiscoveredFile(
                            path=path,
                            relative_path=str(rel_path),
                            file_type=file_type,
                            size_bytes=path.stat().st_size
                        ))
        
        return files
    
    def find_sas7bdat_files(self) -> List[DiscoveredFile]:
        """查找所有 SAS 数据文件"""
        files = []
        
        for path in self.root_dir.rglob("*.sas7bdat"):
            if path.is_file():
                rel_path = path.relative_to(self.root_dir)
                files.append(DiscoveredFile(
                    path=path,
                    relative_path=str(rel_path),
                    file_type='sas7bdat',
                    size_bytes=path.stat().st_size
                ))
        
        return files
    
    def get_file_content(self, file_path: str, max_lines: int = 1000) -> str:
        """获取文件内容（带行数限制）
        
        Args:
            file_path: 文件路径
            max_lines: 最大读取行数
            
        Returns:
            文件内容
        """
        path = Path(file_path)
        if not path.is_absolute():
            path = self.root_dir / file_path
        
        try:
            with open(path, 'r', encoding='utf-8') as f:
```

### 8.2 SAS 分析工具 (sas_analyzer.py)

```python
import re
import json
from pathlib import Path
from typing import Dict, List, Optional, Any
from dataclasses import dataclass, asdict

@dataclass
class TableReference:
    """表引用"""
    table_name: str
    libref: Optional[str] = None
    full_name: str = ""  # libref.table_name

        if not self.full_name and self.libref:

        elif not self.full_name:
            self.full_name = self.table_name

@dataclass
class FieldMapping:
    """字段映射"""
    source_field: str
    target_field: str
    mapping_type: str  # 'passthrough', 'rename', 'calculated'

    def __post_init__(self):
        if self.source_tables is None:
            self.source_tables = []

@dataclass
class SASAnalysisResult:
    """SAS 分析结果"""
    file_path: str
    input_tables: List[TableReference]
    output_tables: List[TableReference]
    field_mappings: List[FieldMapping]
    output_fields: List[str] = None  # KEEP 语句指定的输出字段列表
    output_file: Optional[str] = None  # 输出 CSV 路径
    libname_mappings: Dict[str, str] = None
    
    def __post_init__(self):
        if self.libname_mappings is None:
            self.libname_mappings = {}
        if self.output_fields is None:
            self.output_fields = []
    
    def to_dict(self) -> Dict:
        """转换为字典"""
        return {
            'file_path': self.file_path,
            'input_tables': [asdict(t) for t in self.input_tables],
            'output_tables': [asdict(t) for t in self.output_tables],
            'field_mappings': [asdict(m) for m in self.field_mappings],
            'output_fields': self.output_fields,
            'output_file': self.output_file,
            'libname_mappings': self.libname_mappings
        }
    
    def to_json(self, indent: int = 2) -> str:
        """转换为 JSON"""
        return json.dumps(self.to_dict(), indent=indent, ensure_ascii=False)


class SASAnalyzerTool:
    """SAS 分析工具"""
    
    # 正则表达式模式
    PATTERNS = {
        # DATA 语句
        'data_statement': re.compile(
            r'DATA\s+([a-zA-Z_]\w*(?:\.[a-zA-Z_]\w*)?)\s*;',
            re.IGNORECASE
        ),
        # SET 语句
        'set_statement': re.compile(
            r'SET\s+([a-zA-Z_]\w*(?:\.[a-zA-Z_]\w*)?)(?:\s*\([^)]*\))?\s*;',
            re.IGNORECASE
        ),
        # MERGE 语句
        'merge_statement': re.compile(
            r'MERGE\s+([a-zA-Z_]\w*(?:\s+[a-zA-Z_]\w*(?:\s*\(IN=\w+\))?)*)\s*;',
            re.IGNORECASE
        ),
        # PROC IMPORT
        'proc_import': re.compile(
            r'PROC\s+IMPORT\s+.*?OUT\s*=\s*([a-zA-Z_]\w*)',
            re.IGNORECASE | re.DOTALL
        ),
        # PROC EXPORT
        'proc_export': re.compile(
            r'PROC\s+EXPORT\s+DATA\s*=\s*([a-zA-Z_]\w*).*?OUTFILE\s*=\s*"([^"]+)"',
            re.IGNORECASE | re.DOTALL
        ),
        # LIBNAME
        'libname': re.compile(
            r'LIBNAME\s+([a-zA-Z_]\w*)\s+"([^"]+)"',
            re.IGNORECASE
        ),
        # RENAME
        'rename': re.compile(
            r'RENAME\s*=\s*\(([^)]+)\)',
            re.IGNORECASE
        ),
        # KEEP
        'keep': re.compile(
            r'KEEP\s+([a-zA-Z_]\w*(?:\s+[a-zA-Z_]\w*)*)\s*;',
            re.IGNORECASE
        ),
        # 赋值语句
        'assignment': re.compile(
            r'^\s*([a-zA-Z_]\w*)\s*=\s*([^;]+)\s*;',
            re.IGNORECASE | re.MULTILINE
        )
    }
    
    def analyze_file(self, file_path: str) -> SASAnalysisResult:
        """分析单个 SAS 文件
        
        Args:
            file_path: SAS 文件路径
            
        Returns:
            分析结果
        """
        # 读取文件
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # 提取 libname 映射
        libname_mappings = {}
        for match in self.PATTERNS['libname'].finditer(content):
            libref, path = match.groups()
            libname_mappings[libref] = path
        
        # 提取输入表
        input_tables = []
        
        # SET 语句
        for match in self.PATTERNS['set_statement'].finditer(content):
            table_name = match.group(1)
            input_tables.append(self._parse_table_ref(table_name, libname_mappings))
        
        # MERGE 语句
        for match in self.PATTERNS['merge_statement'].finditer(content):
            merge_clause = match.group(1)
            # 解析多个表名
            table_names = re.findall(r'([a-zA-Z_]\w*)', merge_clause)
            for table_name in table_names:
                if table_name.upper() not in ['IN', 'BY', 'AND', 'OR']:
                    input_tables.append(self._parse_table_ref(table_name, libname_mappings))
        
        # 提取输出表
        output_tables = []
        
        # DATA 语句
        for match in self.PATTERNS['data_statement'].finditer(content):
            table_name = match.group(1)
            output_tables.append(self._parse_table_ref(table_name, libname_mappings))
        
        # 提取输出文件
        output_file = None
        for match in self.PATTERNS['proc_export'].finditer(content):
            output_file = match.group(2)
        
        # 提取字段映射
        field_mappings = self._extract_field_mappings(content, input_tables)
        
        # 提取 KEEP 语句指定的输出字段列表
        output_fields = self._extract_keep_fields(content)
        
        return SASAnalysisResult(
            file_path=file_path,
            input_tables=input_tables,
            output_tables=output_tables,
            field_mappings=field_mappings,
            output_fields=output_fields,
            output_file=output_file,
            libname_mappings=libname_mappings
        )
    
    def _parse_table_ref(
        self, 
        table_name: str, 
        libname_mappings: Dict[str, str]
    ) -> TableReference:
        """解析表引用"""
        if '.' in table_name:
            libref, name = table_name.split('.', 1)
            return TableReference(
                table_name=name,
                libref=libref,
                full_name=table_name
            )
        else:
            # 检查是否在 libname 映射中
            libref = None
            for lib, path in libname_mappings.items():
                if lib.lower() == table_name.lower():
                    libref = lib
                    break
            
            return TableReference(
                table_name=table_name,
                libref=libref,
                full_name=table_name
            )
    
    def _extract_keep_fields(self, content: str) -> List[str]:
        """提取 KEEP 语句指定的输出字段列表
        
        KEEP 语句用于指定 DATA 步输出数据集中保留的字段。
        示例：
            KEEP CustomerId Geography Gender Age Balance credit_scr est_sal;
        
        Args:
            content: SAS 代码内容
            
        Returns:
            字段名列表
        """
        keep_fields = []
        for match in self.PATTERNS['keep'].finditer(content):
            keep_clause = match.group(1)
            # 解析字段名（支持空格分隔）
            fields = re.findall(r'\b([a-zA-Z_]\w*)\b', keep_clause)
            # 过滤掉 SAS 关键字
            keywords = {'AND', 'OR', 'NOT', 'IN', 'EQ', 'NE', 'GT', 'LT', 'GE', 'LE'}
            fields = [f for f in fields if f.upper() not in keywords]
            keep_fields.extend(fields)
        return keep_fields
    
    def _extract_field_mappings(
        self,
        content: str,
        input_tables: List[TableReference]
    ) -> List[FieldMapping]:
        """提取字段映射"""
        mappings = []
        input_table_names = [t.table_name for t in input_tables]
        
        # 提取 RENAME 映射
        for match in self.PATTERNS['rename'].finditer(content):
            rename_clause = match.group(1)
            # 解析 "old = new" 格式
            pairs = re.findall(r'([a-zA-Z_]\w*)\s*=\s*([a-zA-Z_]\w*)', rename_clause)
            for source, target in pairs:
                mappings.append(FieldMapping(
                    source_field=source,
                    target_field=target,
                    mapping_type='rename',
                    formula=f"rename {source} -> {target}",
                    source_tables=input_table_names
                ))
        
        # 提取计算字段
        for match in self.PATTERNS['assignment'].finditer(content):
            target, formula = match.groups()
            target = target.strip()
            formula = formula.strip()
            
            # 跳过 RENAME 中的赋值（已经在上面处理）
            if '=' in formula and 'rename' in formula.lower():
                continue
            
            # 提取公式中引用的字段
            source_fields = re.findall(r'\b([a-zA-Z_]\w*)\b', formula)
            # 过滤掉数字、关键字和 SAS 函数
            keywords = {'if', 'then', 'else', 'do', 'end', 'while', 'for', 
                       'sum', 'mean', 'max', 'min', 'put', 'input', 'substr',
                       'length', 'strip', 'trim', 'left', 'right'}
            source_fields = [
                f for f in source_fields 
                if not f.isdigit() and f.lower() not in keywords
            ]
            
            if source_fields:
                mappings.append(FieldMapping(
                    source_field='; '.join(set(source_fields)),
                    target_field=target,
                    mapping_type='calculated',
                    formula=formula,
                    source_tables=input_table_names
                ))
        
        return mappings
    
    def to_tool_function(self):
        """转换为可调用的工具函数"""
        def analyze_sas_script(file_path: str) -> str:
            """分析 SAS 脚本文件
            
            Args:
                file_path: SAS 文件路径
                
            Returns:
                JSON 格式的分析结果
            """
            result = self.analyze_file(file_path)
            return result.to_json()
        
        return analyze_sas_script
```

### 8.4 血缘存储工具 (lineage_store.py)

```python
import csv
from pathlib import Path
from typing import List, Dict, Optional
from dataclasses import dataclass, asdict
from datetime import datetime

@dataclass
class LineageRecord:
    """血缘记录"""
    output_table: str
    output_column: str
    original_table: str  # 多源用分号分隔
    original_column: str  # 多源用分号分隔
    formula: str
    script_file: str
    created_at: str = ""
    
    def __post_init__(self):
        if not self.created_at:
            self.created_at = datetime.now().isoformat()

class LineageStoreTool:
    """血缘存储工具"""
    
    CSV_COLUMNS = [
        'output_table',
        'output_column', 
        'original_table',
        'original_column',
        'formula',
        'script_file',
        'created_at'
    ]
    
    def __init__(self, output_path: str):
        self.output_path = Path(output_path)
        self.output_path.parent.mkdir(parents=True, exist_ok=True)
        self.records: List[LineageRecord] = []
    
    def add_record(self, record: LineageRecord):
        """添加单条记录"""
        self.records.append(record)
    
    def add_records(self, records: List[LineageRecord]):
        """批量添加记录"""
        self.records.extend(records)
    
    def add_from_dict(self, record_dict: Dict):
        """从字典添加记录"""
        record = LineageRecord(**record_dict)
        self.add_record(record)
    
    def add_from_dicts(self, records_dicts: List[Dict]):
        """从字典列表批量添加记录"""
        for record_dict in records_dicts:
            self.add_from_dict(record_dict)
    
    def export_csv(
        self, 
        columns: Optional[List[str]] = None,
        encoding: str = 'utf-8'
    ) -> Path:
        """导出为 CSV 文件
        
        Args:
            columns: 要导出的列，默认导出所有列
            encoding: 文件编码
            
        Returns:
            输出文件路径
        """
        export_columns = columns or self.CSV_COLUMNS
        
        with open(self.output_path, 'w', newline='', encoding=encoding) as f:
            writer = csv.DictWriter(f, fieldnames=export_columns)
            writer.writeheader()
            
            for record in self.records:
                row = asdict(record)
                # 只写入指定的列
                filtered_row = {k: v for k, v in row.items() if k in export_columns}
                writer.writerow(filtered_row)
        
        return self.output_path
    
    def import_csv(self, file_path: str) -> int:
        """从 CSV 文件导入记录
        
        Args:
            file_path: CSV 文件路径
            
        Returns:
            导入的记录数
        """
        path = Path(file_path)
        if not path.is_absolute():
            path = self.output_path.parent / file_path
        
        count = 0
        with open(path, 'r', encoding='utf-8') as f:
            reader = csv.DictReader(f)
            for row in reader:
                self.add_from_dict(row)
                count += 1
        
        return count
    
    def get_records(self) -> List[LineageRecord]:
        """获取所有记录"""
        return self.records.copy()
    
    def get_records_by_output_table(self, table_name: str) -> List[LineageRecord]:
        """按输出表名筛选记录"""
        return [r for r in self.records if r.output_table == table_name]
    
    def get_records_by_script(self, script_file: str) -> List[LineageRecord]:
        """按脚本文件筛选记录"""
        return [r for r in self.records if script_file in r.script_file]
    
    def count(self) -> int:
        """获取记录数"""
        return len(self.records)
    
    def clear(self):
        """清空所有记录"""
        self.records = []
    
    def to_dict_list(self) -> List[Dict]:
        """转换为字典列表"""
        return [asdict(r) for r in self.records]
    
    def to_json(self, indent: int = 2) -> str:
        """转换为 JSON 字符串"""
        import json
        return json.dumps(self.to_dict_list(), indent=indent, ensure_ascii=False)
    
    def validate(self) -> List[str]:
        """验证所有记录
        
        Returns:
            错误信息列表
        """
        errors = []
        
        for i, record in enumerate(self.records):
            # 检查必填字段
            if not record.output_table:
                errors.append(f"记录 {i}: output_table 不能为空")
            if not record.output_column:
                errors.append(f"记录 {i}: output_column 不能为空")
            if not record.original_table:
                errors.append(f"记录 {i}: original_table 不能为空")
            if not record.original_column:
                errors.append(f"记录 {i}: original_column 不能为空")
            if not record.script_file:
                errors.append(f"记录 {i}: script_file 不能为空")
            
            # 检查多源一致性
            original_tables = record.original_table.split('; ')
            original_columns = record.original_column.split('; ')
            if len(original_tables) != len(original_columns):
                errors.append(
                    f"记录 {i}: 多源表与字段数量不匹配 "
                    f"({len(original_tables)} vs {len(original_columns)})"
                )
        
        return errors
    
    def generate_summary(self) -> str:
        """生成摘要报告"""
        summary_lines = [
            "血缘分析结果摘要",
            "=" * 50,
            f"总记录数：{self.count()}",
            f"输出文件：{self.output_path}",
            ""
        ]
        
        # 按输出表分组统计
        table_counts = {}
        for record in self.records:
            table_counts[record.output_table] = table_counts.get(record.output_table, 0) + 1
        
        summary_lines.append("输出表统计:")
        for table, count in sorted(table_counts.items()):
            summary_lines.append(f"  - {table}: {count} 字段")
        
        return '\n'.join(summary_lines)
```

### B. SAS 代码模式总结

基于 `/input/sas/` 目录下的示例代码，识别出以下常见模式：

1. **单表转换**（01_churn_transform.sas, 02_loan_transform.sas）：
   - SET 语句读取单表
   - RENAME 重命名字段
   - 计算派生字段
   - KEEP 选择输出字段

2. **多表合并**（03_merge_transform.sas）：
   - 多个 DATA 步骤准备中间表
   - PROC SORT 排序
   - MERGE 语句合并多表
   - 跨表计算字段

3. **数据集定义**（def_Churn.sas, def_train.sas）：
   - libname 映射
   - infile 读取原始数据
   - input 语句定义字段格式

|------|------|

| Rename | 字段重命名 |
| Calculated | 通过计算表达式派生的字段 |
